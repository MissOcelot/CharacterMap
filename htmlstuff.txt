<script>
network.on("click", function(params) {
  if(params.nodes.length === 0) return;

  let selected = params.nodes[0];
  let connected = network.getConnectedNodes(selected);
  connected.push(selected);

  // Show only clicked node + neighbors
  network.body.data.nodes.get().forEach(n => {
    network.body.data.nodes.update({id:n.id, hidden: !connected.includes(n.id)});
  });

  network.body.data.edges.get().forEach(e => {
    let fromSelected = e.from === selected;
    let toSelected = e.to === selected;
    network.body.data.edges.update({id:e.id, hidden: !(fromSelected || toSelected)});
  });

  // --- Update info panel ---
  const panel = document.getElementById("infoPanel");
  const rels = relationships[selected] || [];
  if (rels.length === 0) {
    panel.innerHTML = "<b>" + selected + "</b><br>No listed connections.";
  } else {
    let html = "<b>" + selected + "</b><br><ul>";
    rels.forEach(r => {
      html += "<li><b>" + r.name + "</b>: " + r.type + "</li>";
    });
    html += "</ul>";
    panel.innerHTML = html;
  }
  panel.style.display = "block"; // show panel
});
</script>

function showFullMap() {
  // Reset all nodes and edges to visible
  network.body.data.nodes.get().forEach(n => network.body.data.nodes.update({id:n.id, hidden:false}));
  network.body.data.edges.get().forEach(e => network.body.data.edges.update({id:e.id, hidden:false}));
}

network.on("click", function(params) {
  if(params.nodes.length === 0) return;

  let selected = params.nodes[0];
  let connected = network.getConnectedNodes(selected);
  connected.push(selected);

  // Show only the clicked node + its neighbors
  network.body.data.nodes.get().forEach(n => {
    network.body.data.nodes.update({id:n.id, hidden: !connected.includes(n.id)});
  });

  // Show only edges that touch the selected node (sever other connections)
  network.body.data.edges.get().forEach(e => {
    let fromSelected = e.from === selected;

});
